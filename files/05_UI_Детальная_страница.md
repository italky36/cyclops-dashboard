# Задание 5: UI — Детальная страница сделки

## Контекст
Проект Cyclops Dashboard. Список сделок готов. Нужно создать страницу просмотра и управления конкретной сделкой.

## Входные данные
- Хуки: `hooks/useDeals.ts` (хук `useDeal`)
- Утилиты: `lib/utils/deals.ts`
- Типы: `types/cyclops/deals.ts`

## Задача
Создать страницу детальной сделки: `app/(dashboard)/deals/[dealId]/page.tsx`

## Структура страницы

### 1. Шапка
- Кнопка "← Назад к списку"
- ID сделки (полный)
- Внешний ключ (ext_key) — если есть
- Статус (Badge)
- Кнопки действий (зависят от статуса)

### 2. Блок "Основная информация"
- Общая сумма
- Дата создания
- Дата обновления

### 3. Блок "Плательщики"
Таблица:
| Виртуальный счёт | Сумма | Исполнено |
|------------------|-------|-----------|

- Виртуальный счёт — ссылка на `/virtual-accounts/[id]`
- Исполнено — галочка/крестик/прочерк

Если есть документы:
- Список документов с типом и статусом добавления

### 4. Блок "Получатели"
Таблица:
| № | Тип | Сумма | Статус | Платёж | Ошибка |
|---|-----|-------|--------|--------|--------|

- Тип — человекочитаемое название
- Статус — Badge (new/in_process/reject/success)
- Платёж — ID платежа, ссылка на `/payments/[id]`
- Ошибка — красным, если есть error_reason

Раскрывающиеся реквизиты (по клику на строку):
- Все поля из requisites
- Для payment_contract: счёт, БИК, ИНН, КПП, название, назначение
- Для СБП: ФИО, телефон, банк
- Для карты: (номер скрыт)

### 5. Кнопки действий

Логика отображения по статусу:

| Статус | Исполнить | Редактировать | Отменить | Отменить из коррекции | Комплаенс |
|--------|-----------|---------------|----------|----------------------|-----------|
| new | ✅ | ✅ | ✅ | ❌ | ✅ |
| in_process | ❌ | ❌ | ❌ | ❌ | ❌ |
| partial | ✅ (частично) | ✅ | ❌ | ❌ | ✅ |
| correction | ❌ | ✅ | ❌ | ✅ | ✅ |
| closed | ❌ | ❌ | ❌ | ❌ | ❌ |
| rejected | ❌ | ❌ | ❌ | ❌ | ❌ |

### 6. Модальные окна

**Подтверждение исполнения:**
- Текст: "Вы уверены, что хотите исполнить сделку?"
- Для partial: чекбоксы выбора получателей
- Кнопки: "Отмена" / "Исполнить"

**Подтверждение отмены:**
- Текст: "Сделка будет отменена. Это действие необратимо."
- Кнопки: "Отмена" / "Подтвердить"

**Результат комплаенс-проверки:**
- Список получателей с результатами
- approved: true → зелёная галочка
- approved: false → красный крестик + messages

## Реализация

### Файл: `app/(dashboard)/deals/[dealId]/page.tsx`

```tsx
'use client';

import { useState } from 'react';
import { useRouter } from 'next/navigation';
import Link from 'next/link';
import { useDeal } from '@/hooks/useDeals';
import {
  DEAL_STATUS_LABELS,
  DEAL_STATUS_COLORS,
  RECIPIENT_TYPE_LABELS,
  formatAmount,
  formatDate,
  getAvailableActions,
} from '@/lib/utils/deals';
import type { DealRecipientInfo, ComplianceCheckResult } from '@/types/cyclops/deals';

// Статус получателя
const RECIPIENT_STATUS_COLORS = {
  new: 'bg-gray-100 text-gray-800',
  in_process: 'bg-blue-100 text-blue-800',
  reject: 'bg-red-100 text-red-800',
  success: 'bg-green-100 text-green-800',
  old: 'bg-gray-100 text-gray-500',
};

const RECIPIENT_STATUS_LABELS = {
  new: 'Новый',
  in_process: 'В процессе',
  reject: 'Отклонён',
  success: 'Успешно',
  old: 'Устаревший',
};

// Компонент модального окна
function Modal({ 
  open, 
  onClose, 
  title, 
  children 
}: { 
  open: boolean; 
  onClose: () => void; 
  title: string; 
  children: React.ReactNode;
}) {
  if (!open) return null;
  
  return (
    <div className="fixed inset-0 bg-black bg-opacity-50 flex items-center justify-center z-50">
      <div className="bg-white rounded-lg p-6 max-w-md w-full mx-4">
        <h3 className="text-lg font-semibold mb-4">{title}</h3>
        {children}
      </div>
    </div>
  );
}

// Компонент реквизитов получателя
function RecipientRequisites({ requisites }: { requisites: Record<string, any> }) {
  const fields = [
    { key: 'account', label: 'Счёт' },
    { key: 'bank_code', label: 'БИК' },
    { key: 'name', label: 'Наименование' },
    { key: 'inn', label: 'ИНН' },
    { key: 'kpp', label: 'КПП' },
    { key: 'purpose', label: 'Назначение' },
    { key: 'first_name', label: 'Имя' },
    { key: 'last_name', label: 'Фамилия' },
    { key: 'middle_name', label: 'Отчество' },
    { key: 'phone_number', label: 'Телефон' },
    { key: 'bank_sbp_id', label: 'Банк СБП' },
    { key: 'virtual_account', label: 'Виртуальный счёт' },
  ];

  return (
    <div className="bg-gray-50 p-4 rounded-lg mt-2">
      <div className="grid grid-cols-2 gap-2 text-sm">
        {fields.map(({ key, label }) => 
          requisites[key] ? (
            <div key={key}>
              <span className="text-gray-500">{label}:</span>{' '}
              <span className="font-mono">{requisites[key]}</span>
            </div>
          ) : null
        )}
      </div>
    </div>
  );
}

export default function DealPage({ params }: { params: { dealId: string } }) {
  const router = useRouter();
  const { dealId } = params;
  
  const { 
    deal, 
    loading, 
    error, 
    executeDeal, 
    rejectDeal, 
    cancelFromCorrection, 
    checkCompliance 
  } = useDeal(dealId);

  const [expandedRecipient, setExpandedRecipient] = useState<number | null>(null);
  const [executeModalOpen, setExecuteModalOpen] = useState(false);
  const [rejectModalOpen, setRejectModalOpen] = useState(false);
  const [cancelModalOpen, setCancelModalOpen] = useState(false);
  const [complianceResult, setComplianceResult] = useState<ComplianceCheckResult[] | null>(null);
  const [selectedRecipients, setSelectedRecipients] = useState<number[]>([]);
  const [actionLoading, setActionLoading] = useState(false);

  if (loading) {
    return (
      <div className="p-6">
        <div className="animate-pulse space-y-4">
          <div className="h-8 bg-gray-200 rounded w-1/3"></div>
          <div className="h-40 bg-gray-200 rounded"></div>
          <div className="h-60 bg-gray-200 rounded"></div>
        </div>
      </div>
    );
  }

  if (error || !deal) {
    return (
      <div className="p-6">
        <div className="bg-red-50 text-red-600 p-4 rounded-lg">
          {error || 'Сделка не найдена'}
        </div>
        <button 
          onClick={() => router.push('/deals')}
          className="mt-4 text-blue-600 hover:underline"
        >
          ← Вернуться к списку
        </button>
      </div>
    );
  }

  const actions = getAvailableActions(deal.status);

  // Обработчики действий
  const handleExecute = async () => {
    setActionLoading(true);
    try {
      const recipients = selectedRecipients.length > 0 ? selectedRecipients : undefined;
      await executeDeal(recipients);
      setExecuteModalOpen(false);
      setSelectedRecipients([]);
    } catch (e) {
      // Ошибка уже в error
    } finally {
      setActionLoading(false);
    }
  };

  const handleReject = async () => {
    setActionLoading(true);
    try {
      await rejectDeal();
      setRejectModalOpen(false);
    } catch (e) {
      // Ошибка уже в error
    } finally {
      setActionLoading(false);
    }
  };

  const handleCancelFromCorrection = async () => {
    setActionLoading(true);
    try {
      await cancelFromCorrection();
      setCancelModalOpen(false);
    } catch (e) {
      // Ошибка уже в error
    } finally {
      setActionLoading(false);
    }
  };

  const handleCheckCompliance = async () => {
    setActionLoading(true);
    try {
      const result = await checkCompliance();
      setComplianceResult(result);
    } catch (e) {
      // Ошибка уже в error
    } finally {
      setActionLoading(false);
    }
  };

  return (
    <div className="p-6">
      {/* Шапка */}
      <div className="mb-6">
        <button 
          onClick={() => router.push('/deals')}
          className="text-blue-600 hover:underline mb-2"
        >
          ← К списку сделок
        </button>
        
        <div className="flex justify-between items-start">
          <div>
            <h1 className="text-2xl font-bold mb-2">
              Сделка
            </h1>
            <p className="text-sm text-gray-500 font-mono mb-1">{deal.id}</p>
            {deal.ext_key && (
              <p className="text-sm text-gray-500">
                Внешний ключ: <span className="font-mono">{deal.ext_key}</span>
              </p>
            )}
          </div>
          
          <span className={`px-3 py-1 rounded-full text-sm font-medium ${DEAL_STATUS_COLORS[deal.status]}`}>
            {DEAL_STATUS_LABELS[deal.status]}
          </span>
        </div>
      </div>

      {/* Кнопки действий */}
      <div className="flex gap-2 mb-6">
        {actions.canExecute && (
          <button
            onClick={() => setExecuteModalOpen(true)}
            className="px-4 py-2 bg-green-600 text-white rounded-md hover:bg-green-700"
          >
            Исполнить
          </button>
        )}
        
        {actions.canEdit && (
          <button
            onClick={() => router.push(`/deals/${dealId}/edit`)}
            className="px-4 py-2 bg-blue-600 text-white rounded-md hover:bg-blue-700"
          >
            Редактировать
          </button>
        )}
        
        {actions.canReject && (
          <button
            onClick={() => setRejectModalOpen(true)}
            className="px-4 py-2 bg-red-600 text-white rounded-md hover:bg-red-700"
          >
            Отменить
          </button>
        )}
        
        {actions.canCancelFromCorrection && (
          <button
            onClick={() => setCancelModalOpen(true)}
            className="px-4 py-2 bg-orange-600 text-white rounded-md hover:bg-orange-700"
          >
            Отменить из коррекции
          </button>
        )}
        
        {(actions.canExecute || actions.canEdit) && (
          <button
            onClick={handleCheckCompliance}
            disabled={actionLoading}
            className="px-4 py-2 bg-gray-600 text-white rounded-md hover:bg-gray-700 disabled:opacity-50"
          >
            Проверить комплаенс
          </button>
        )}
      </div>

      {/* Результат комплаенс */}
      {complianceResult && (
        <div className="bg-gray-50 p-4 rounded-lg mb-6">
          <h3 className="font-semibold mb-2">Результат проверки комплаенс</h3>
          {complianceResult.map((item) => (
            <div key={item.number} className="flex items-start gap-2 mb-2">
              <span className={item.approved ? 'text-green-600' : 'text-red-600'}>
                {item.approved ? '✓' : '✗'}
              </span>
              <div>
                <span>Получатель #{item.number}</span>
                {item.messages.map((msg, i) => (
                  <p key={i} className={`text-sm ${msg.level === 'ERROR' ? 'text-red-600' : 'text-yellow-600'}`}>
                    [{msg.level}] {msg.text}
                  </p>
                ))}
              </div>
            </div>
          ))}
          <button 
            onClick={() => setComplianceResult(null)}
            className="text-sm text-gray-500 hover:underline mt-2"
          >
            Скрыть
          </button>
        </div>
      )}

      {/* Основная информация */}
      <div className="bg-white rounded-lg shadow p-6 mb-6">
        <h2 className="text-lg font-semibold mb-4">Основная информация</h2>
        <div className="grid grid-cols-3 gap-4">
          <div>
            <p className="text-sm text-gray-500">Общая сумма</p>
            <p className="text-xl font-semibold">{formatAmount(deal.amount)}</p>
          </div>
          <div>
            <p className="text-sm text-gray-500">Создана</p>
            <p>{formatDate(deal.created_at)}</p>
          </div>
          <div>
            <p className="text-sm text-gray-500">Обновлена</p>
            <p>{formatDate(deal.updated_at)}</p>
          </div>
        </div>
      </div>

      {/* Плательщики */}
      <div className="bg-white rounded-lg shadow p-6 mb-6">
        <h2 className="text-lg font-semibold mb-4">Плательщики</h2>
        <table className="min-w-full">
          <thead>
            <tr className="border-b">
              <th className="text-left py-2 text-sm text-gray-500">Виртуальный счёт</th>
              <th className="text-left py-2 text-sm text-gray-500">Сумма</th>
              <th className="text-left py-2 text-sm text-gray-500">Исполнено</th>
            </tr>
          </thead>
          <tbody>
            {deal.payers.map((payer, index) => (
              <tr key={index} className="border-b">
                <td className="py-3">
                  <Link 
                    href={`/virtual-accounts/${payer.virtual_account}`}
                    className="text-blue-600 hover:underline font-mono text-sm"
                  >
                    {payer.virtual_account.slice(0, 8)}...
                  </Link>
                </td>
                <td className="py-3">{formatAmount(payer.amount)}</td>
                <td className="py-3">
                  {payer.executed === true ? '✓' : payer.executed === false ? '✗' : '—'}
                </td>
              </tr>
            ))}
          </tbody>
        </table>
      </div>

      {/* Получатели */}
      <div className="bg-white rounded-lg shadow p-6">
        <h2 className="text-lg font-semibold mb-4">Получатели</h2>
        <table className="min-w-full">
          <thead>
            <tr className="border-b">
              <th className="text-left py-2 text-sm text-gray-500">№</th>
              <th className="text-left py-2 text-sm text-gray-500">Тип</th>
              <th className="text-left py-2 text-sm text-gray-500">Сумма</th>
              <th className="text-left py-2 text-sm text-gray-500">Статус</th>
              <th className="text-left py-2 text-sm text-gray-500">Платёж</th>
              <th className="text-left py-2 text-sm text-gray-500">Ошибка</th>
            </tr>
          </thead>
          <tbody>
            {deal.recipients.map((recipient) => (
              <>
                <tr 
                  key={recipient.number}
                  onClick={() => setExpandedRecipient(
                    expandedRecipient === recipient.number ? null : recipient.number
                  )}
                  className="border-b hover:bg-gray-50 cursor-pointer"
                >
                  <td className="py-3">{recipient.number}</td>
                  <td className="py-3">{RECIPIENT_TYPE_LABELS[recipient.type] || recipient.type}</td>
                  <td className="py-3">{formatAmount(recipient.amount)}</td>
                  <td className="py-3">
                    <span className={`px-2 py-1 rounded-full text-xs ${RECIPIENT_STATUS_COLORS[recipient.status]}`}>
                      {RECIPIENT_STATUS_LABELS[recipient.status]}
                    </span>
                  </td>
                  <td className="py-3">
                    {recipient.payment ? (
                      <Link 
                        href={`/payments/${recipient.payment.id}`}
                        className="text-blue-600 hover:underline text-sm font-mono"
                        onClick={(e) => e.stopPropagation()}
                      >
                        {recipient.payment.id.slice(0, 8)}...
                      </Link>
                    ) : '—'}
                  </td>
                  <td className="py-3 text-red-600 text-sm">
                    {recipient.error_reason || ''}
                  </td>
                </tr>
                {expandedRecipient === recipient.number && (
                  <tr>
                    <td colSpan={6}>
                      <RecipientRequisites requisites={recipient.requisites} />
                    </td>
                  </tr>
                )}
              </>
            ))}
          </tbody>
        </table>
      </div>

      {/* Модальные окна */}
      <Modal
        open={executeModalOpen}
        onClose={() => setExecuteModalOpen(false)}
        title="Исполнение сделки"
      >
        {deal.status === 'partial' ? (
          <div className="mb-4">
            <p className="mb-2">Выберите получателей для исполнения:</p>
            {deal.recipients
              .filter(r => r.status === 'new' || r.status === 'reject')
              .map(r => (
                <label key={r.number} className="flex items-center gap-2 mb-1">
                  <input
                    type="checkbox"
                    checked={selectedRecipients.includes(r.number)}
                    onChange={(e) => {
                      if (e.target.checked) {
                        setSelectedRecipients([...selectedRecipients, r.number]);
                      } else {
                        setSelectedRecipients(selectedRecipients.filter(n => n !== r.number));
                      }
                    }}
                  />
                  #{r.number} — {RECIPIENT_TYPE_LABELS[r.type]} — {formatAmount(r.amount)}
                </label>
              ))
            }
          </div>
        ) : (
          <p className="mb-4">Вы уверены, что хотите исполнить сделку?</p>
        )}
        <div className="flex gap-2 justify-end">
          <button
            onClick={() => setExecuteModalOpen(false)}
            className="px-4 py-2 bg-gray-200 rounded-md"
          >
            Отмена
          </button>
          <button
            onClick={handleExecute}
            disabled={actionLoading}
            className="px-4 py-2 bg-green-600 text-white rounded-md disabled:opacity-50"
          >
            {actionLoading ? 'Выполняется...' : 'Исполнить'}
          </button>
        </div>
      </Modal>

      <Modal
        open={rejectModalOpen}
        onClose={() => setRejectModalOpen(false)}
        title="Отмена сделки"
      >
        <p className="mb-4 text-red-600">
          Сделка будет отменена. Это действие необратимо.
        </p>
        <div className="flex gap-2 justify-end">
          <button
            onClick={() => setRejectModalOpen(false)}
            className="px-4 py-2 bg-gray-200 rounded-md"
          >
            Отмена
          </button>
          <button
            onClick={handleReject}
            disabled={actionLoading}
            className="px-4 py-2 bg-red-600 text-white rounded-md disabled:opacity-50"
          >
            {actionLoading ? 'Выполняется...' : 'Подтвердить отмену'}
          </button>
        </div>
      </Modal>

      <Modal
        open={cancelModalOpen}
        onClose={() => setCancelModalOpen(false)}
        title="Отмена из коррекции"
      >
        <p className="mb-4">
          Сделка будет отменена. Уже исполненные платежи останутся без изменений.
        </p>
        <div className="flex gap-2 justify-end">
          <button
            onClick={() => setCancelModalOpen(false)}
            className="px-4 py-2 bg-gray-200 rounded-md"
          >
            Отмена
          </button>
          <button
            onClick={handleCancelFromCorrection}
            disabled={actionLoading}
            className="px-4 py-2 bg-orange-600 text-white rounded-md disabled:opacity-50"
          >
            {actionLoading ? 'Выполняется...' : 'Подтвердить'}
          </button>
        </div>
      </Modal>
    </div>
  );
}
```
